# Use the latest Alpine image with Python
FROM python:3.14.0a2-alpine3.20 AS base

# Set environment variable for Flask
ENV FLASK_APP=flaskr

# Define the PATH target
ENV PATH="/opt/gtk/bin:$PATH"

# Set the working directory
WORKDIR /flaskr

# Copy the application files
COPY flaskr /flaskr

# Copy the requirements file
COPY requirements.txt /flaskr

# Install dependencies
RUN pip install --no-cache-dir -r requirements.txt

# Stage 2: Install Trivy
FROM aquasec/trivy:latest AS trivy

# Copy the application files from the base stage
COPY --from=base /flaskr /flaskr

# Run Trivy scan and generate JSON report
RUN trivy image --format json --output /cve_analysis_results.json python:3.14.0a2-alpine3.20

# Stage 3: Final image
FROM python:3.14.0a2-alpine3.20

# Set the working directory
WORKDIR /flaskr

# Copy the application files from the base stage
COPY --from=base /flaskr /flaskr

# Copy the Trivy scan results
COPY --from=trivy /flaskr/cve_analysis_results.json /flaskr/cve_analysis_results.json

# Expose the port the app runs on
EXPOSE 5000

# Define the command to run the application
CMD ["flask", "run", "--host=0.0.0.0", "--port=5000"]
